import { render, screen, fireEvent, waitFor } from "@testing-library/react";
import VolunteerMatching from "../pages/volunteerMatching";
 
// Mock fetch for API calls
global.fetch = jest.fn();
 
const mockVolunteers = [
  {
    id: "v1",
    name: "Volunteer Name",
    skills: ["deploy", "code"],
    preferences: "night"
  },
  {
    id: "v2",
    name: "Jane Smith",
    skills: ["design", "teaching"],
    preferences: "weekend"
  }
];
 
const mockMatchedEvents = [
  {
    id: "e1",
    name: "Blood Drive",
    description: "Saving lives",
    skills: ["packing", "assisting"],
    date: "2025-03-15",
    location: "Community Center",
    urgency: "high",
    matchScore: 85
  }
];
 
const mockAssignResponse = { success: true };
 
beforeEach(() => {
  fetch.mockReset();
});
 
describe("VolunteerMatching Component", () => {
  test("renders the component correctly", () => {
    render(<VolunteerMatching />);
    expect(screen.getByText(/Volunteer App/i)).toBeInTheDocument();
    expect(screen.getByText(/Welcome, admin/i)).toBeInTheDocument();
  });
 
  test("fetches and displays volunteers", async () => {
    fetch.mockResolvedValueOnce({
      json: jest.fn().mockResolvedValueOnce({ success: true, volunteers: mockVolunteers })
    });
 
    render(<VolunteerMatching />);
    await waitFor(() => expect(screen.getByText("Volunteer Name")).toBeInTheDocument());
    expect(screen.getByText("Jane Smith")).toBeInTheDocument();
  });
 
  test("handles volunteer selection and matches events", async () => {
    fetch.mockResolvedValueOnce({
      json: jest.fn().mockResolvedValueOnce({ success: true, volunteers: mockVolunteers })
    });
 
    render(<VolunteerMatching />);
 
    await waitFor(() => screen.getByText("Volunteer Name"));

    fireEvent.change(screen.getByRole("combobox"), { target: { value: "v1" } });

    fetch.mockResolvedValueOnce({
      json: jest.fn().mockResolvedValueOnce({ success: true, volunteer: mockVolunteers[0], matchedEvents: mockMatchedEvents })
    });

    await waitFor(() => screen.getByText("Blood Drive"));
    expect(screen.getByText(/Saving lives/i)).toBeInTheDocument();
  });
 
  test("handles assignment of a volunteer to an event", async () => {
    fetch.mockResolvedValueOnce({
      json: jest.fn().mockResolvedValueOnce({ success: true, volunteers: mockVolunteers })
    });
 
    render(<VolunteerMatching />);
 
    await waitFor(() => screen.getByText("Volunteer Name"));

    fireEvent.change(screen.getByRole("combobox"), { target: { value: "v1" } });

    fetch.mockResolvedValueOnce({
      json: jest.fn().mockResolvedValueOnce({ success: true, volunteer: mockVolunteers[0], matchedEvents: mockMatchedEvents })
    });

    await waitFor(() => screen.getByText("Blood Drive"));

    fetch.mockResolvedValueOnce({
      json: jest.fn().mockResolvedValueOnce(mockAssignResponse)
    });

    fireEvent.click(screen.getByText(/Assign Volunteer/i));

    await waitFor(() => expect(screen.getByText(/Successfully assigned/i)).toBeInTheDocument());
  });
 
  test("displays error message if assignment fails", async () => {
    fetch.mockResolvedValueOnce({
      json: jest.fn().mockResolvedValueOnce({ success: true, volunteers: mockVolunteers })
    });
 
    render(<VolunteerMatching />);
 
    await waitFor(() => screen.getByText("Volunteer Name"));

    fireEvent.change(screen.getByRole("combobox"), { target: { value: "v1" } });

    fetch.mockResolvedValueOnce({
      json: jest.fn().mockResolvedValueOnce({ success: true, volunteer: mockVolunteers[0], matchedEvents: mockMatchedEvents })
    });

    await waitFor(() => screen.getByText("Blood Drive"));

    fetch.mockResolvedValueOnce({
      json: jest.fn().mockResolvedValueOnce({ success: false })
    });

    fireEvent.click(screen.getByText(/Assign Volunteer/i));

    await waitFor(() => expect(screen.getByText(/Cannot assign/i)).toBeInTheDocument());
  });
 
  test("handles API error and falls back to hardcoded data", async () => {
    fetch.mockRejectedValueOnce(new Error("API error"));
 
    render(<VolunteerMatching />);
 
    await waitFor(() => expect(screen.getByText("Volunteer Name")).toBeInTheDocument());
    expect(screen.getByText("Jane Smith")).toBeInTheDocument();
  });
});
